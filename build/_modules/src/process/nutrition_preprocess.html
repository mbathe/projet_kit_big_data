def clean_data(df):
    """
    Nettoie les donnÃ©es en filtrant les recettes ayant trop peu de notes et en supprimant les valeurs aberrantes

    Returns:

    """

    logger.info("Nettoyage des donnÃ©es...")

    # Filtrer les donnÃ©es (nombre de notes >= 5)

    # Supprimer les valeurs nutritionnelles aberrantes
    nutrition_columns = ['Graisses', 'Graisse_saturÃ©es',
                         'Sucre', 'Sodium', 'ProtÃ©ines', 'Glucides']
    for column in nutrition_columns:

    df = df.drop(df[df[('Calories')] > 800].index)
    logger.info("Valeurs aberrantes supprimÃ©es pour la colonne Calories.")

    logger.info("Nettoyage des donnÃ©es terminÃ©.")

    return df
